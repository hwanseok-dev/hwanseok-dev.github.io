## 첫 번째 풀이

### 알고리즘

struct를 사용한 풀이로 완전탐색으로 진행합니다. 

### 정답 코드

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

struct cake {
	int x, y, z;
};
int main()
{
	//freopen("input.txt", "r", stdin);
	ios_base::sync_with_stdio(0); cin.tie(0);

	
	cake a, b, c;
	cin >> a.x >> a.y >> a.z;
	cin >> c.x >> c.y >> c.z;

	for (int i = 1; i <= 100; i++) {
		for (int j = 1; j <= 100; j++) {
			for (int k = 1; k <= 100; k++) {
				if (a.z + i == c.x &&
					a.y * j == c.y &&
					a.x + k == c.z) {
					cout << i << " " << j << " " << k << "\n";
					return 0;
				}
			}
		}
	}

	return 0;
}
```
